<!DOCTYPE html>

<html>

    <head>

        <script src='https://api.tiles.mapbox.com/mapbox-gl-js/v2.2.0/mapbox-gl.js'></script>

        <link href='https://api.tiles.mapbox.com/mapbox-gl-js/v2.2.0/mapbox-gl.css' rel='stylesheet' />

        <style>

            /* Add some styling for the map container */

            #map {

                height: 400px;

                width: 100%;

            }

        </style>

    </head>

    <body>

        <div id='map'></div>

        <form>

            <input id='address' type='text' placeholder='Enter an address'/>

            <button id='search'>Search</button>

        </form>

        <script>

            // Initialize the map

            mapboxgl.accessToken = 'your_access_token';

            var map = new mapboxgl.Map({

                container: 'map',

                style: 'mapbox://styles/mapbox/streets-v11',

                center: [-98.4916, 29.4252],

                zoom: 12

            });

            // Add the geocoder control

            var geocoder = new MapboxGeocoder({

                accessToken: mapboxgl.accessToken,

                mapboxgl: mapboxgl

            });

            map.addControl(geocoder);

            // Add a listener for the 'result' event

            geocoder.on('result', function(ev) {

                var coordinates = ev.result.geometry.coordinates;

                // Create and add the marker to the map

                var marker = new mapboxgl.Marker()

                    .setLngLat(coordinates)

                    .addTo(map);

            });

            //Search button event listener

            document.getElementById('search').addEventListener('click', function(e) {

                e.preventDefault();

                geocoder.query(document.getElementById('address').value);

            });

        </script>

    </body>

</html>
